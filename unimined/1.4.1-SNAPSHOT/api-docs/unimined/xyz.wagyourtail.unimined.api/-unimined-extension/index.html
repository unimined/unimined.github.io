abstract class UniminedExtension(val project: Project) : FabricLikeApiExtension

the main entrypoint.

unimined.useGlobalCache = false

    // See MinecraftConfig
}

// optional second config for another sourceSet
unimined.minecraft(sourceSets.other) {
    // See MinecraftConfig
}
